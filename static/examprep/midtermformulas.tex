		\item $ESS = \displaystyle \sum^n_{i=1} (\hat{Y_i}-\bar{Y})^2$
		\item $TSS = \displaystyle \sum^n_{i=1} (Y_i-\bar{Y})^2$
		\item $SER = \sigma_u= \sqrt{\displaystyle\frac{SSE}{n-2}}$
